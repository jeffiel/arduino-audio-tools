102  static void task(void *pvParameters) {
103  TRACED();
104  static uint8_t buffer[512];
105  BufferedTaskStream *self = (BufferedTaskStream *)pvParameters;
106  while (true) {
107  size_t available_to_write = self->buffers.availableForWrite();
108  if (*(self->p_stream) && available_to_write > 0) {
109  size_t to_read = min(available_to_write, (size_t)512);
110  size_t avail_read =
111  self->p_stream->readBytes((uint8_t *)buffer, to_read);
112  size_t written = self->buffers.writeArray(buffer, avail_read);
113 
114  if (written != avail_read) {
115  LOGE("DATA Lost! %zu reqested, %zu written!", avail_read, written);
116  }
117 
118  } else {
119  // 3ms at 44100 stereo is about 529.2 bytes
120  delay(3);
121  }
122  // buffer is full we start to provide data

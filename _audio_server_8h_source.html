216  // Handle an new client connection and return the data
217  void processClient() {
218  // LOGD("processClient");
219  if (client_obj) { // if you get a client,
220  LOGI("New Client."); // print a message out the serial port
221  String currentLine =
222  ""; // make a String to hold incoming data from the client
223  while (client_obj.connected()) { // loop while the client's connected
224  if (client_obj
225  .available()) { // if there's bytes to read from the client,
226  char c = client_obj.read(); // read a byte, then
227  if (c == '\n') { // if the byte is a newline character
228 
229  // if the current line is blank, you got two newline characters in a
230  // row. that's the end of the client HTTP request, so send a
231  // response:
232  if (currentLine.length() == 0) {
233  sendReplyHeader();
234  sendReplyContent();
235  // break out of the while loop:
236  break;
237  } else { // if you got a newline, then clear currentLine:
238  currentLine = "";
239  }
240  } else if (c != '\r') { // if you got anything else but a carriage
241  // return character,
242  currentLine += c; // add it to the end of the currentLine
243  }
244  }
245  }
246  }

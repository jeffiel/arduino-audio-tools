225  bool beginCodec(I2SCodecConfig info) {
226  codec_cfg.sd_active = info.sd_active;

232  codec_cfg.i2s.rate = toRate(info.sample_rate);
233  codec_cfg.i2s.fmt = toFormat(info.i2s_format);
234  // use reverse logic for codec setting

236  if (p_board == nullptr) return false;

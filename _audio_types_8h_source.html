386  uint8_t *p_result = (uint8_t*) data;
387  int open = samples*sizeof(T);
388  int total = 0;
389  // copy missing data

391  int to_write = min(open, static_cast<int>(maxSamples*sizeof(T)));
392  int written = p_out->write(p_result+total, to_write );
393  open -= written;
394  total += written;

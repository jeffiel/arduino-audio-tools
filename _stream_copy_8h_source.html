98  if (!active) return 0;
99  // if not initialized we do nothing
100  if (from==nullptr && to==nullptr) return 0;
101 
102  // E.g. if we try to write to a server we might not have any output destination yet
103  int to_write = to->availableForWrite();
104  if (check_available_for_write && to_write==0){
105  delay(500);
106  return 0;
107  }
108 
109  size_t result = 0;
110  size_t delayCount = 0;
111  size_t len = buffer_size;
112  if (check_available) {
113  len = available();
114  }
115  size_t bytes_to_read = buffer_size;
116  size_t bytes_read = 0;
117 
118  if (len > 0){
119  bytes_to_read = min(len, static_cast<size_t>(buffer_size));
120  size_t samples = bytes_to_read / sizeof(T);
121  bytes_to_read = samples * sizeof(T);
122  // don't overflow buffer
123  if (to_write > 0){
124  bytes_to_read = min((int)bytes_to_read, to_write);
125  }
126 
127  // get the data now
128  bytes_read = 0;
129  if (bytes_to_read>0){
130  bytes_read = from->readBytes((uint8_t*)&buffer[0], bytes_to_read);

132

134  notifyMime(buffer.data(), bytes_to_read);
135 
136  // write data
137  result = write(bytes_read, delayCount);
138 
139  // callback with unconverted data
140  if (onWrite!=nullptr) onWrite(onWriteObj, &buffer[0], result);

330  size_t write(size_t len, size_t &delayCount ){
331  if (!buffer || len==0) return 0;
332  LOGD("write: %d", (int)len);
333  size_t total = 0;
334  long open = len;
335  int retry = 0;
336  while(open > 0){
337  size_t written = to->write((const uint8_t*)buffer.data()+total, open);
338  LOGD("write: %d -> %d", (int) open, (int) written);
339  total += written;
340  open -= written;
341  delayCount++;
342 
343  if (open > 0){
344  // if we still have progress we reset the retry counter
345  if (written>0) retry = 0;
346 
347  // abort if we reached the retry limit
348  if (retry++ > retryLimit){
349  LOGE("write %s to target has failed after %d retries! (%ld bytes)", log_name, retry, open);
350  break;
351  }
352 
353  // wait a bit
354  if (retry>1) {
355  delay(retry_delay);
356  LOGI("try write %s - %d (open %ld bytes) ",log_name, retry, open);
357  }
358  }
359 
360  CHECK_MEMORY();
361  }
362  return total;
363  }

366  void notifyMime(void* data, size_t len){
367  if (is_first && len>4) {
368  const uint8_t *start = (const uint8_t *) data;
369  actual_mime = "audio/basic";
370  if (start[0]==0xFF && start[1]==0xF1){
371  actual_mime = "audio/aac";
372  } else if (memcmp(start,"ID3",3) || start[0]==0xFF || start[0]==0xFE ){
373  actual_mime = "audio/mpeg";
374  } else if (memcmp(start,"RIFF",4)){
375  actual_mime = "audio/vnd.wave";
376  }
377  if (notifyMimeCallback!=nullptr){
378  notifyMimeCallback(actual_mime);
379  }
380  }
381  is_first = false;
